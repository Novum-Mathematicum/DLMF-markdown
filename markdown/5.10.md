$$
\operatorname{Ln}\Gamma\left(z\right)+z-\left(z-\tfrac{1}{2}\right)\ln z-\tfrac{1}{2}\ln\left(2\pi\right)=\cfrac{a_{0}}{z+\cfrac{a_{1}}{z+\cfrac{a_{2}}{z+\cfrac{a_{3}}{z+\cfrac{a_{4}}{z+\cfrac{a_{5}}{z+}}}}}}\cdots, \tag{5.10.1}
$$
